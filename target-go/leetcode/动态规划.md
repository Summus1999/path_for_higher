# 动态规划

## Easy

#### 70.爬楼梯

```c++
class Solution {
public:
    int climbStairs(int n) {
        if (n <= 1) {
            return 1;
        }
        int dp_i_minus_2 = 1;
        int dp_i_minus_1 = 1;
        int ans = 0;
        // 从第2步开始迭代计算到第n步
        for (int i = 2; i <= n; i++) {
            ans = dp_i_minus_2 + dp_i_minus_1;
            dp_i_minus_2 = dp_i_minus_1;
            dp_i_minus_1 = ans;
        }
        return ans;
    }
};
```



## Medium

#### 120.三角形最小路径和

```c++
class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        int n = triangle.size();
        // 存储动态规划过程中的状态
        auto f = triangle[n - 1];
        // 由顶到底的最短路径取决于它下方相邻的两个位置
        for (int i = n - 2; i >= 0; i--) {
            for (int j = 0; j <= i; j++) {
                f[j] = min(f[j], f[j + 1]) + triangle[i][j];
            }
        }
        return f[0];
    }
};
```
