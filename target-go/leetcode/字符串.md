#### 5.最长回文子串

```c++
class Solution {
public:
    string longestPalindrome(string s) {
        if (s.empty()) {
            return "";
        }
        // 预处理
        string t = "^";
        for (auto cur : s) {
            t += '#';
            t += cur;
        }
        t += "#$";
        int n = t.size();
        vector<int> p(n, 0);
        int center = 0;
        int right = 0;
        int maxCenter = 0;
        int maxLength = 0;
        // 遍历处理后的字符串 t 的每一个字符（除了首尾两个特殊字符）
        for (int i = 1; i < n - 1; i++) {
            if (i < right) {
                p[i] = min(right - i, p[2 * center - i]);
            } else {
                p[i] = 1;
            }
            // 尝试向两边扩展
            while (t[i + p[i]] == t[i - p[i]]) {
                p[i]++;
            }
            if (i + p[i] > right) {
                center = i;
                right = i + p[i];
            }
            // 更新记录的最长回文串信息
            if (p[i] > maxLength) {
                maxLength = p[i];
                maxCenter = i;
            }
        }
        // 计算原字符串中对应最长回文子串的起始位置和长度，并返回结果
        int start = (maxCenter - maxLength) / 2;
        return s.substr(start, maxLength - 1);
    }
};
```
