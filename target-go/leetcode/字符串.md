# 字符串

## Easy

#### 415.字符串相加

```c++
class Solution {
public:
    string addStrings(string num1, string num2) {
        int m = num1.size();
        int n = num2.size();
        // i 和 j 分别指向两个字符串的末尾，从低位开始相加
        int i = m - 1;
        int j = n - 1;
        int add = 0;
        string ans;
        ans.reserve(max(m, n) + 1);
        // 当 num1、num2 还有位数未处理，或者仍有进位时，继续循环
        while (i >= 0 || j >= 0 || add) {
            auto x = (i >= 0) ? num1[i--] - '0' : 0;
            auto y = (j >= 0) ? num2[j--] - '0' : 0;
            int sum = x + y + add;
            ans.push_back('0' + sum % 10);
            add = sum / 10;
        }
        return string(ans.rbegin(), ans.rend());
    }
};
```



## Medium

#### 5.最长回文子串

```c++
class Solution {
public:
    string longestPalindrome(string s) {
        if (s.empty()) {
            return "";
        }
        // 预处理
        string t = "^";
        for (auto cur : s) {
            t += '#';
            t += cur;
        }
        t += "#$";
        int n = t.size();
        vector<int> p(n, 0);
        int center = 0;
        int right = 0;
        int maxCenter = 0;
        int maxLength = 0;
        // 遍历处理后的字符串 t 的每一个字符（除了首尾两个特殊字符）
        for (int i = 1; i < n - 1; i++) {
            if (i < right) {
                p[i] = min(right - i, p[2 * center - i]);
            } else {
                p[i] = 1;
            }
            // 尝试向两边扩展
            while (t[i + p[i]] == t[i - p[i]]) {
                p[i]++;
            }
            if (i + p[i] > right) {
                center = i;
                right = i + p[i];
            }
            // 更新记录的最长回文串信息
            if (p[i] > maxLength) {
                maxLength = p[i];
                maxCenter = i;
            }
        }
        // 计算原字符串中对应最长回文子串的起始位置和长度，并返回结果
        int start = (maxCenter - maxLength) / 2;
        return s.substr(start, maxLength - 1);
    }
};
```
