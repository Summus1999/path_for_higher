
# 二分查找

## Easy

#### 704.二分查找

```c++
class Solution {
public:
    int search(vector<int>& nums, int target) {
        // 二分查找模板题
        int left = 0;
        int right = nums.size() - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                return mid;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }
};
```

#### 69.x的平方根

```c++
class Solution {
public:
    int mySqrt(int x) {
        // 特殊情况处理
        if (x == 0) {
            return 0;
        }
        int left = 1;
        int right = min(x, 46340);
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (mid * mid <= x) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return right;
    }
};
```



## Medium

#### 50.Pow(x,n)

```c++
class Solution {
public:
    double myPow(double x, int n) {
        // 使用快速幂算法
        long long N = n;
        if (N < 0) {
            x = 1 / x;
            N = -N;
        }
        double ans = 1.0;
        double base = x;
        while (N > 0) {
            if (N % 2 == 1) {
                ans *= base;
            }
            base *= base;
            N /= 2;
        }
        return ans;
    }
};
```
