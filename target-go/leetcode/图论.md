#### 200.岛屿数量

```c++
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int m=grid.size();
        int n=grid[0].size();
        int ans=0;
        // DFS：将当前连通陆地('1')全部标记为'2'（已访问），避免重复计数
        auto dfs=[&](this auto&&dfs,int i,int j)->void{
            if(i<0||i>=m||j<0||j>=n||grid[i][j]!='1'){
                return;
            }
            // 标记当前陆地为已访问
            grid[i][j]='2';
            dfs(i,j+1);
            dfs(i,j-1);
            dfs(i-1,j);
            dfs(i+1,j);
        };
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]=='1'){
                    // 遇到未访问岛屿，启动DFS并计数
                    dfs(i,j);
                    ans++;
                }
            }
        }
        return ans;
    }
};
```
