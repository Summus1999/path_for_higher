## Easy

#### 1013.将数组分成和相等的三个部分

```c++
class Solution {
public:
    bool canThreePartsEqualSum(vector<int>& arr) {
        int total = accumulate(arr.begin(), arr.end(), 0);
        if (total % 3 != 0) {
            return false;
        }
        int target = total / 3;
        int findCount = 0;
        int curSum = 0;
        for (auto cur : arr) {
            curSum += cur;
            if (curSum == target) {
                findCount++;
                // 重置，开始下一段
                curSum = 0;
            }
        }
        return findCount >= 3;
    }
};
```

#### 283.移动零

```c++
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int n = nums.size();
        int write = 0;
        // 原地覆盖非零元素
        for (int read = 0; read < n; read++) {
            if (nums[read] != 0) {
                nums[write] = nums[read];
                write++;
            }
        }
        // 原地将剩余位置置为0
        while (write < n) {
            nums[write] = 0;
            write++;
        }
    }
};

```
