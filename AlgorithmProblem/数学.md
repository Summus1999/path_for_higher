# 数学

## leetcode 1056 易混淆数

```c++
class Solution {
public:
    bool confusingNumber(int n) {
        int rotateMap[10] = {0, 1, -1, -1, -1, -1, 9, -1, 8, 6};
        int originalNum = n;
        int processedNum = 0;
        /*逐位处理数字*/
        while (n > 0) {
            int digit = n % 10;
            n /= 10;
            if (rotateMap[digit] == -1) {
                return false;
            }
            processedNum = processedNum * 10 + rotateMap[digit];
        }
        /*如果数字与原来的数字不等则为true*/
        return processedNum != originalNum;
    }
};
```
